-- ---------------------------------------------------------------------------
-- 1. Configuration
-- ---------------------------------------------------------------------------
set INTERFACE to "en0"
set LOOP_COUNT to 500

-- ---------------------------------------------------------------------------
-- 2. Select the keyword list file and define the output directory
-- ---------------------------------------------------------------------------
try
	set selectedFile to choose file with prompt "Select keyword.txt:"
	set KEYWORD_FILE_PATH to POSIX path of selectedFile
	tell application "Finder" to set parentPath to POSIX path of ((container of selectedFile) as alias)
	set OUTPUT_DIR_PATH to parentPath & "pcap_file"
on error
	return
end try

-- Automatically resolve the local tshark executable.
set tsharkPath to ""
if (do shell script "[ -f /Applications/Wireshark.app/Contents/MacOS/tshark ] && echo 'yes' || echo 'no'") is "yes" then
	set tsharkPath to "/Applications/Wireshark.app/Contents/MacOS/tshark"
else
	set tsharkPath to "/usr/local/bin/tshark"
end if

display dialog "Using network interface: " & INTERFACE & return & "Enter the administrator password to run tshark:" default answer "" with hidden answer
set adminPass to text returned of result

-- Create the output directory and load the keyword list.
do shell script "mkdir -p " & quoted form of OUTPUT_DIR_PATH
set keywordList to paragraphs of (read (KEYWORD_FILE_PATH as POSIX file) as «class utf8»)

log "=== STARTING DATA COLLECTION ==="

-- Wrap the full workflow in a 10-minute timeout to reduce AppleEvent timeout failures.
with timeout of 600 seconds
	
	-- -----------------------------------------------------------------------
	-- 3. Main collection loop
	-- -----------------------------------------------------------------------
	repeat with aKeyword in keywordList
		if (length of aKeyword) > 0 then
			
			-- Build a filesystem-safe keyword token to avoid Unicode normalization
			-- mismatches (e.g., NFD/NFC on macOS) and duplicate filename issues.
			set safeKeyword to do shell script "echo " & quoted form of aKeyword & " | iconv -t ASCII//TRANSLIT | tr -d '\\r' | sed 's/[^a-zA-Z0-9]/_/g' | sed 's/__*/_/g' | sed 's/^_//' | sed 's/_$//'"
			
			log "=== KEYWORD: " & aKeyword & " (Filename: " & safeKeyword & ") ==="
			
			-- Resume logic: skip the keyword if the final expected capture already exists.
			set fileMaxPath to OUTPUT_DIR_PATH & "/" & safeKeyword & "_" & LOOP_COUNT & ".pcap"
			set isFullyDone to do shell script "[ -f " & quoted form of fileMaxPath & " ] && echo 'yes' || echo 'no'"
			
			if isFullyDone is "yes" then
				log " -> [SKIP] Already collected " & LOOP_COUNT & " traces for this keyword."
			else
				
				repeat with loopIndex from 1 to LOOP_COUNT
					
					set filename to safeKeyword & "_" & loopIndex & ".pcap"
					set pcapPath to OUTPUT_DIR_PATH & "/" & filename
					
					set fileExists to do shell script "[ -f " & quoted form of pcapPath & " ] && echo 'yes' || echo 'no'"
					
					if fileExists is "yes" then
						-- Skip iterations whose capture file is already present.
					else
						log "   > Running iteration: " & loopIndex
						
						-- Verify Internet connectivity before starting a new capture.
						log "     [NET] Checking Internet connectivity..."
						repeat
							try
								do shell script "ping -c 1 -W 2000 8.8.8.8"
								log "     [NET] -> Connectivity confirmed."
								exit repeat
							on error
								log "     [NET] -> Connection unavailable. Retrying in 5 seconds..."
								delay 5
							end try
						end repeat
						
						-- A. Start tshark capture in the background.
						set startCmd to "echo " & quoted form of adminPass & " | sudo -S " & quoted form of tsharkPath & " -i " & INTERFACE & " -w " & quoted form of pcapPath & " > /dev/null 2>&1 &"
						try
							do shell script startCmd
						end try
						
						-- B. Open Safari and issue the query.
						set the clipboard to (aKeyword as text)
						
						-- Force Safari to launch and move to the foreground.
						do shell script "open -a Safari"
						tell application "Safari" to activate
						
						-- [Delay 1] Allow 1.5 s for the Safari process to initialize.
						delay (3 / 2)
						
						-- Enforce focus on Safari to avoid pasting into the script editor.
						tell application "System Events"
							set frontmost of process "Safari" to true
							repeat 20 times
								if frontmost of process "Safari" then exit repeat
								delay (1 / 10)
							end repeat
						end tell
						
						tell application "Safari"
							try
								if (count of windows) is 0 then
									make new document
								end if
							end try
						end tell
						
						-- [Delay 2] Wait 0.5 s before synthesizing keystrokes.
						delay (1 / 2)
						
						-- Send keystrokes only if Safari is confirmed to be frontmost.
						tell application "System Events"
							if frontmost of process "Safari" then
								tell process "Safari"
									try
										keystroke "l" using command down
										delay (1 / 2) -- [Delay 3] 0.5 s
										keystroke "v" using command down
										delay (1 / 2) -- [Delay 4] 0.5 s
										key code 36 -- Enter
									end try
								end tell
							else
								log "     [WARN] Safari focus was lost. Keystrokes were skipped to protect the script."
							end if
						end tell
						
						-- C. Wait for the Google results page to load.
						-- [Delay 5] Reserve 7 s for the search results page.
						log "       -> Query submitted. Waiting 7 seconds for the Google results page..."
						delay 7
						
						-- D. Perform controlled DOM-based redirection to the first result,
						-- handling cookie-consent popups when present.
						log "       -> Extracting the first result link and redirecting..."
						tell application "Safari"
							try
								set jsScript to "
								(function() {
									function tryClick() {
										var btns = document.querySelectorAll('button');
										for (var i=0; i<btns.length; i++) {
											if (btns[i].innerText.match(/Accept|Agree|Allow|Reject|Decline/i)) {
												btns[i].click();
												return 'clicked_cookie';
											}
										}
										
										var headers = document.querySelectorAll('h3');
										for (var i = 0; i < headers.length; i++) {
											var link = headers[i].closest('a');
											if (link && link.href) {
												var href = link.href;
												if (href.includes('google.com/search') || href.includes('google.com.vn/search')) continue;
												if (href.startsWith('/search')) continue;
												if (href.includes('googleadservices')) continue;
												
												link.click();
												return 'clicked_link';
											}
										}
										return 'not_found';
									}
									return tryClick();
								})();"
								
								repeat 10 times
									set clickResult to do JavaScript jsScript in current tab of window 1
									if clickResult is "clicked_link" then
										exit repeat
									else if clickResult is "clicked_cookie" then
										delay 2
									else
										delay 1
									end if
								end repeat
							on error errMsg
								log "JavaScript error: " & errMsg
							end try
						end tell
						
						-- E. Allow the destination page to generate network traffic.
						-- [Delay 6] Reserve 15 s for page-load traffic collection.
						log "       -> Redirect complete. Waiting 15 seconds for destination-page traffic..."
						delay 15
						
						-- F. Cleanup: terminate capture, close Safari, and clear local state.
						try
							do shell script "echo " & quoted form of adminPass & " | sudo -S kill -9 $(pgrep tshark)"
						end try
						
						try
							tell application "Safari" to quit
						end try
						
						tell application "System Events"
							repeat 20 times
								if not (exists process "Safari") then exit repeat
								delay (1 / 2)
							end repeat
						end tell
						
						try
							do shell script "rm -f ~/Library/Safari/History.db*"
							do shell script "rm -f ~/Library/Cookies/Cookies.binarycookies"
							do shell script "rm -rf ~/Library/Caches/com.apple.Safari"
							do shell script "rm -rf ~/Library/Saved\\ Application\\ State/com.apple.Safari.savedState"
						end try
						
						-- [Delay 7] Allow 2 s for cleanup to settle.
						log "       -> Clearing local browser state (2 seconds)..."
						delay 2
						
						-- [Delay 8] Add a random 2-4 s inter-trace pause.
						set interDelay to (random number from 2 to 4)
						log "       -> Idle interval: " & interDelay & " s"
						delay interDelay
						
					end if -- End if fileExists
				end repeat
			end if -- End if isFullyDone
		end if
	end repeat
	
end timeout

display dialog "DATA COLLECTION COMPLETED." buttons {"OK"} default button "OK"
